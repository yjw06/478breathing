## Global Constraints

- 단일 파일 `index.html` 하나로 완결. 외부 파일·CDN·네트워크 요청 0개.
- 오디오 파일을 번들하지 않는다. 모든 소리는 런타임 합성(`SpeechSynthesis` / `Web Audio`).
- 음성 언어는 한국어(`ko-KR`) 단일. 한국어 음성 부재 시 기본 음성 폴백.
- 모든 오디오는 사용자 제스처("시작" 탭) 이후에만 시작(iOS 자동재생 정책).

- 야간 사용: 배경은 거의 검정, 표시는 흐릿하게, 터치 타깃은 크게.
- 빌드 도구·테스트 프레임워크 없음. 로직은 순수 함수로 분리해 콘솔 어서션으로 검증하고, 통합 동작은 브라우저에서 수동 검증한다.

```js
const DEFAULTS = { inhale:4, hold:7, exhale:8, cycles:4 };
const KEY = "breath378.settings";

function clampSettings(s) {
  const n = (v, min, max, def) => {
    const x = Math.round(Number(v));
    return Number.isFinite(x) ? Math.min(max, Math.max(min, x)) : def;
  };
  return {
    inhale: n(s.inhale, 1, 20, DEFAULTS.inhale),
    hold:   n(s.hold,   0, 30, DEFAULTS.hold),
    exhale: n(s.exhale, 1, 30, DEFAULTS.exhale),
    cycles: n(s.cycles, 1, 20, DEFAULTS.cycles),
  };
}
```

```js
function buildSchedule(s) {
  const steps = [{ phase:"prep", label:"준비", durationMs:3000, cycleIndex:0 }];
  for (let c = 0; c < s.cycles; c++) {
    steps.push({ phase:"inhale", label:"들이쉬기", durationMs:s.inhale*1000, cycleIndex:c });
    if (s.hold > 0) steps.push({ phase:"hold", label:"멈추기", durationMs:s.hold*1000, cycleIndex:c });
    steps.push({ phase:"exhale", label:"내쉬기", durationMs:s.exhale*1000, cycleIndex:c });
  }
  steps.push({ phase:"done", label:"완료", durationMs:0, cycleIndex:s.cycles });
  return steps;
}

// --- temp assertions ---
const sch = buildSchedule({inhale:4,hold:7,exhale:8,cycles:4});
console.assert(sch[0].phase === "prep", "starts prep");
console.assert(sch[sch.length-1].phase === "done", "ends done");
console.assert(sch.filter(x=>x.phase==="inhale").length === 4, "4 inhales");
console.assert(sch.filter(x=>x.phase==="hold").length === 4, "4 holds");
console.assert(sch[1].durationMs === 4000, "inhale 4s");
console.assert(buildSchedule({inhale:4,hold:0,exhale:8,cycles:2}).filter(x=>x.phase==="hold").length === 0, "hold=0 skipped");
console.log("Task3 assertions done");
```

- [ ] **Step 2: 타임스탬프 기반 러너 구현**

```js
function runSession(schedule, onPhase, onDone) {
  let i = 0, timer = null, stopped = false;
  function next() {
    if (stopped) return;
    if (i >= schedule.length) { onDone(); return; }
    const step = schedule[i++];
    onPhase(step, schedule);
    if (step.phase === "done") { onDone(); return; }
    const dueAt = performance.now() + step.durationMs;
    (function tick() {
      if (stopped) return;
      const remaining = dueAt - performance.now();
      if (remaining <= 0) next();
      else timer = setTimeout(tick, Math.min(remaining, 200)); // drift-corrected
    })();
  }
  next();
  return function stop() { stopped = true; if (timer) clearTimeout(timer); };
}
```

```js
let audioCtx = null, noiseNode = null, noiseGain = null;
function getAudioCtx() {
  if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();
  if (audioCtx.state === "suspended") audioCtx.resume();
  return audioCtx;
}
function startNoise() {
  const ctx = getAudioCtx();
  const bufferSize = 2 * ctx.sampleRate;
  const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
  const data = buffer.getChannelData(0);
  let last = 0;
  for (let i = 0; i < bufferSize; i++) {           // brown noise
    const white = Math.random() * 2 - 1;
    last = (last + 0.02 * white) / 1.02; data[i] = last * 3.5;
  }
  noiseNode = ctx.createBufferSource(); noiseNode.buffer = buffer; noiseNode.loop = true;
  const lp = ctx.createBiquadFilter(); lp.type = "lowpass"; lp.frequency.value = 500;
  noiseGain = ctx.createGain();
  noiseGain.gain.setValueAtTime(0, ctx.currentTime);
  noiseGain.gain.linearRampToValueAtTime(0.06, ctx.currentTime + 1.5); // gentle fade-in
  noiseNode.connect(lp).connect(noiseGain).connect(ctx.destination);
  noiseNode.start();
}
function stopNoise() {
  if (!noiseNode || !audioCtx) return;
  const t = audioCtx.currentTime;
  noiseGain.gain.cancelScheduledValues(t);
  noiseGain.gain.setValueAtTime(noiseGain.gain.value, t);
  noiseGain.gain.linearRampToValueAtTime(0, t + 1);
  const n = noiseNode; setTimeout(() => { try { n.stop(); } catch {} }, 1100);
  noiseNode = null;
}
```
